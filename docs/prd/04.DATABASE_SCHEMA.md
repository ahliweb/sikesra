# Database Schema — SIKESRA MVP

**Platform**: Cloudflare D1 (SQLite at edge)  
**Namespace**: Plugin-owned (`sikesra_*`)  
**Migration**: Idempoten via install/activate hook

## Migration Pattern (Install Hook)

```typescript
// src/runtime.ts — install hook
async function installPlugin(ctx: PluginContext) {
  // Idempoten: CREATE TABLE IF NOT EXISTS
  await ctx.db.run(CREATE_REGISTRY_ENTITIES_SQL)
  await ctx.db.run(CREATE_VERIFICATION_EVENTS_SQL)
  // ... semua tabel

  // Seed default data (INSERT OR IGNORE)
  await ctx.db.run(SEED_PERMISSIONS_SQL)
  await ctx.db.run(SEED_ROLES_SQL)
  await ctx.db.run(SEED_ROLE_PERMISSIONS_SQL)
  await ctx.db.run(SEED_ABAC_ATTRIBUTES_SQL)
  await ctx.db.run(SEED_ABAC_POLICIES_SQL)

  // Record install event
  const record = createAuditRecord({
    kind: "plugin.installed",
    scope: "sikesra:lifecycle",
    actor: "system",
    summary: "Plugin SIKESRA installed and storage initialized",
    metadata: {}
  })
  await ctx.storage.auditEvents.create(record)
}
```
